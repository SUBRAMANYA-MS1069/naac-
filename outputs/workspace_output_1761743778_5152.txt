7. Database Schema Guidelines
7.1 Required Fields in Every Collection

 javascript

 {
     // Primary Key - ALWAYS custom string ID
     [entity]Id: {
      type: String,
      default: () => uuidv4(),
      unique: true,
      required: true,
      index: true
     },


     // Tenant isolation - ALWAYS required
     tenantId: {
      type: String,
      required: true,
      index: true,
      ref: 'Tenant'
     },


     // Soft delete - ALWAYS include
     isActive: {
      type: Boolean,
      default: true,
      index: true
     },
